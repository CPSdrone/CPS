Les symboles ajoutés sont
+
*
#
°
°C regroupé un un seul caractère et qu'il faut appeller en tapant ^  dans vos programmes 
Fléche montante en tapant {
Flèche descendante en tapant}
Flèche horizontale en tapant ¨ (tréma)
%

byte convert_ascii (int character)
{
// Transposition de la table ASCII (non standard) integrée à l'EEPROM du MAX7456
// f5mna_6.mcm
 byte lookup_char;

  if (character == 32)
    lookup_char = 0x00; // blank space
  else if (character == 48)
    lookup_char = 0x0a; // 0
  else if ((character > 48) && (character < 58))
    lookup_char = (character - 48); // 1-9
  else if ((character > 64) && (character < 91))
    lookup_char = (character - 54); // A-Z
  else if ((character > 96) && (character < 123))
    lookup_char = (character - 60); // a-z
  else if (character == 34)
    lookup_char = 0x48; // "
  else if (character == 39)
    lookup_char = 0x46; // '
  else if (character == 40)
    lookup_char = 0x3f; // (
  else if (character == 41)
    lookup_char = 0x40; // )
  else if (character == 44)
    lookup_char = 0x45; // ,
  else if (character == 45)
    lookup_char = 0x49; // -
  else if (character == 46)
    lookup_char = 0x41; // .
  else if (character == 47)
    lookup_char = 0x47; // /
  else if (character == 58)
    lookup_char = 0x44; // :
  else if (character == 59)
    lookup_char = 0x43; // ;
  else if (character == 60)
    lookup_char = 0x4a; // <
  else if (character == 62)
    lookup_char = 0x4b; // >
  else if (character == 63)
    lookup_char = 0x42; // ?
  else if (character == 64)
    lookup_char = 0x4c; // @
 //Modif tableau Std:  f5mna_3.mcm  
 //*********************************   
  else if (character == 43)
    lookup_char = 0x4e; // + 
  else if (character == 42)
    lookup_char = 0x4d ; // *    
  else if (character == 35)
    lookup_char = 0x51 ; // #    
  else if (character == 176)
    lookup_char = 0x4f ; // °     
  else if (character == 94)//^
    lookup_char = 0x50 ; // °C
 //********************************* 
 //    FLECHES  f5mna_4.mcm
 //********************************
  else if (character == 123) //Fléche montante
    lookup_char = 0x52 ;     // {  
  else if (character == 125) //Flèche descendante
    lookup_char = 0x53 ;     //}     
  else if (character == 168) //Flèche horizontale
    lookup_char = 0x54 ;     // ¨ tréma
 //********************************* 
 //    Pourcent  f5mna_6.mcm
 //********************************
  else if (character == 37)  // %
  lookup_char = 0x55 ;       // %
